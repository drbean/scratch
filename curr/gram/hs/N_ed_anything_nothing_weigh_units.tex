\newcommand{\myContent}[0]{

\setlength{\leftmargini}{-0.5cm}

\begin{description}

\item [adjective + N-ed]\

\begin{itemize*}
\item A man/woman with a big/small/yellow nose/body/head.
\item A man/woman with big/small/yellow ears/eyes/legs.
\item A house/building/town with a tall/three/red roof/chimney/window.
\end{itemize*}

\item [Anything, nothing]\

\begin{description}
\item Anything but small = Very, very big
\item Anything but safe = Very, very unsafe
\item Anything but a good driver = A very, very bad driver
\end{description}

\begin{description}
\item Vegetarians eat nothing but vegetables (and rice, etc).
\item Be careful. The bad driver ahead of us is nothing but a menace. 
\item I tell salespeople who come to my door to go away. They are nothing but a nuisance. 
\end{description}

\item [weigh]\

\begin{description}
\item Did Meow the cat weigh 22 kilograms?
\item Can you lift a weight of 22 kilograms?
\item How much did/does/do sb/sth weigh?
\item How much did Meow weigh?
\end{description}

\item [hundred, thousand, million]\
	\begin{itemize*}
	\item 2,800,000
	\item 2,018,000
	\item 280,000
	\item 201,800
	\end{itemize*}

\end{description}
}
