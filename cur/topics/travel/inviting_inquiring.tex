\begin{Large}Inviting someone to the movies\end{Large}

\begin{multicols}{2}

\begin{description}

\item [Checking availability]\

\begin{itemize*}
\item What are you doing Saturday night?
\item Do you have any plans Saturday night?
\item Are you busy on Saturday night?
\end{itemize*}

\item [Invitation]\

\begin{itemize*}
\item What about going to the movies on Saturday night?
\item Do you want to (Would you like to) go to the movies on Saturday night?
\item Let's go to the movies on Saturday night.
\item Come with me to the movies on Saturday night.
\end{itemize*}

\item [Making an excuse]\

\begin{itemize*}
\item Sorry, I can't. I have to wash my hair.
\item I wish I could, but I have to study for an exam.
\end{itemize*}

\item [Responding to a refusal]\
\begin{itemize*}
\item That's OK. Some other time, perhaps.
\item Too bad. OK. I'll find someone else.
\end{itemize*}

\end{description}

\end{multicols}

\begin{Large}Checking the movie start time\end{Large}

\begin{multicols}{2}

\begin{description}

\item [Calling the movie theater]\

\begin{itemize*}
\item Hello? Is that Apex Movie Theater?
\end{itemize*}

\item [Asking about the movie]\

\begin{itemize*}
\item I want to see the movie, Terminator III.
\item Is Terminator III still showing?
\end{itemize*}

\item [Asking about the start time]\

\begin{itemize*}
\item What time does the movie start?
\end{itemize*}

\item [Booking a seat]\
\begin{itemize*}
\item Can I book a seat?
\end{itemize*}

\end{description}

\end{multicols}

}
