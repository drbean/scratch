\begin{itemize*}
\item [WH questions]%

\begin{large}WH-phrase + auxiliary verb + subject + verb (dictionary form) + the rest\end{large}
\end{itemize*}

% \begin{multicols}{1}

\begin{itemize*}
  \item What kind of supermarket do you like/dislike?
  \item Where do you usually go to the supermarket?
  \item What was the last supermarket you went to?
  \item How often do you go to the supermarket?
  \item How much money do you spend when you go to the supermarket?
  \item Who goes grocery shopping in your family?
\end{itemize*}
\begin{itemize*}
  \item [WH-word] Who/What/When/Where/Why/How?
  \item [Auxiliary verb] do/does/did, is/are/was/were, can, etc
  \item [Subject] you/your family/you and your friends/the cashier
  \item [Verb phrase] spend, go grocery shopping, look for, buy
\end{itemize*}

\begin{itemize*}
\item [YN questions]%

\begin{large}auxiliary verb + subject + verb (dictionary form) + the rest\end{large}
\end{itemize*}

\begin{itemize*}
  \item Have you ever been to (supermarket name) in (place name)?
  \item Do you ever .... supermarkets?
  \item Do you like going to the supermarket?
  \item Do you choose a supermarket for the goods or the prices?
\end{itemize*}

\begin{itemize*}
\item [Tag questions]%

\begin{large}sentence + tag\end{large}
\begin{large}sentence = subject + verb + the rest\end{large}
\begin{large}tag = auxiliary verb + pronoun\end{large}
\end{itemize*}

\begin{itemize*}
  \item You go to 全聯, don't you?
  \item You have been to Carrefour, haven't you?
  \item You were at 7-11 yesterday, weren't you?
\end{itemize*}
