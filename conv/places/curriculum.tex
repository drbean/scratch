\newcommand{\myContent}[0]{

\setlength{\leftmargini}{-0.5cm}

\begin{description}

\item [Questions about going places]\

\begin{itemize*}
\item Where have you been (to) in Taiwan?
\item Do you like to travel?
\item Do you want to travel overseas? Where?
\end{itemize*}

\item [Destination guessing game]\

\begin{description}
\item One person thinks of a destination. The others try to guess the destination by asking YN questions. Is it in the north of Taiwan? Is it on the (East/West) coast? Is it inland/in the mountains? Is it famous for a lake? Is there a train station there?
\end{description}

\item [The 3 types of traveler]\

\begin{description}
\item Tourists, backpackers, expatriates
\end{description}

\item [Taiwan destinations]\
	\begin{description*}
	\item [Cities] \indent Taipei, Hsinchu, Taichung, Tainan, Kaohsiung
	\item [Seaports and the coast] Keelung, Tamsui, Baishatun, Tongxiao, Yuanli
	\item [Mountains] Arishan, Sun Moon Lake, Cross-Island Highway
	\end{description*}

\item [Directions]\
	\begin{enumerate}
	\item Q: How do I get to the train station (from here?)
	\item Q: I'm looking for a convenience store.
	\item Q: Where's the nearest public convenience (BrE)/restroom (AmE)?
	\item Q: Which way is North?
	\item A: Turn left at the second set of traffic lights and walk for about 200 meters. It's on the right.
	\end{enumerate}

\end{description}
}
